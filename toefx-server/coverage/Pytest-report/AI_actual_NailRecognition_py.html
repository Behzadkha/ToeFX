40 Finds each nail in an image and returns them in a list.

42 @:return: A list of nail images for each nail found in the original image.

53 # Get the score and class label for each potential nail in the image

55 classesTensor = NailRecognition.model.get_tensor_by_name("detection_classes:0") 

56 numDetections = NailRecognition.model.get_tensor_by_name("num_detections:0")

58 # Load the image

62 output = [] # A list of nail images

64 # Process the image 

65 image = NailRecognition.IsolateHand(image.copy())

69 # Find the nails in the image 

70 (boundaries, scores, labels, N) = session.run([boxesTensor, scoresTensor, 

71 classesTensor, numDetections],

86 endY = int(endY * height)

88 output.append(baseImage[startY:endY, startX:endX]) # Crop to the nail

95 def IsolateHand(image): 

96 """ 

97 Applies skin colour binarization to the image. Anything that isn't skin colour is blotted out. 

98 :param image: The image to process. 

99 :return: A new image with the binarization applied. 

100 """ 

101 

102 YCrCb_frame = cv2.cvtColor(image, cv2.COLOR_BGR2YCrCb) 

103 YCrCb_frame = cv2.GaussianBlur(YCrCb_frame, (3, 3), 0) 

104 binMask = cv2.inRange(YCrCb_frame, np.array([0, 127, 75]), np.array([255, 177, 130])) 

105 

106 kernel = cv2.getStructuringElement(cv2.MORPH_ELLIPSE, (5, 5)) 

107 binMask = cv2.dilate(binMask, kernel, iterations=5) 

108 result = cv2.bitwise_and(image, image, mask=binMask) 

109 

110 return result
